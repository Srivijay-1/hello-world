# Function to create a simple bar chart of the data
def show_chart():
    import matplotlib.pyplot as plt

    # Set the style to use Times New Roman font
    plt.rcParams['font.family'] = 'Times New Roman'

    # Create a figure and axis
    fig, ax = plt.subplots(figsize=(12, 6))

    # Plot the data
    x = range(len(df['Country']))
    width = 0.2

    ax.bar([i - 1.5*width for i in x], df['Overall Account Ownership (%)'], width, label='Overall', color='#1a5276')
    ax.bar([i - 0.5*width for i in x], df['Male Account Ownership (%)'], width, label='Male', color='#2e86c1')
    ax.bar([i + 0.5*width for i in x], df['Female Account Ownership (%)'], width, label='Female', color='#5dade2')
    ax.bar([i + 1.5*width for i in x], df['Youth Account Ownership (%)'], width, label='Youth', color='#aed6f1')

    # Add labels and title
    ax.set_xlabel('Region', fontsize=12)
    ax.set_ylabel('Account Ownership (%)', fontsize=12)
    ax.set_xticks(x)
    ax.set_xticklabels(df['Country'], rotation=45, ha='right')

    # Add a legend
    ax.legend()

    # Add grid lines
    ax.grid(True, linestyle='--', alpha=0.7)

    # Adjust layout
    plt.tight_layout()

    return fig
